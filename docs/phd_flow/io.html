339    def remove(
340        self, key: PATH_LIKE, local: bool = True, remote: bool = False
341    ) -> None:
342        if remote and not local:
343            raise ValueError(
344                "It is not a good idea to remove remote files "
345                "while keeping the local files. "
346                f"Key: {key}"
347            )
348
349        if remote:
350            logger.info(f"Deleting remote: {key}")
351            for fid, _ in self.bucket.ls(str(self.remote_path / key)):
352                fid.delete()
353        if local and (self / key).exists():
354            logger.info(f"Deleting local: {key}")

374    def sync(self, source: str, destination: str) -> None:
375        logger.info(f"Sync: {source} -> {destination}")
376        source = b2_api.parse_sync_folder(source, self.b2)
377        destination = b2_api.parse_sync_folder(destination, self.b2)
378        for folder in [source, destination]:
379            if isinstance(folder, b2_api.LocalFolder):
380                os.makedirs(folder.root, exist_ok=True)
381
382        policies_manager = b2_api.ScanPoliciesManager(exclude_all_symlinks=True)
383
384        synchronizer = b2_api.Synchronizer(
385            max_workers=10,
386            policies_manager=policies_manager,
387            dry_run=False,
388            allow_empty_source=True,
389        )
390
391        no_progress = False
392
393        with b2_api.SyncReport(sys.stdout, no_progress) as reporter:
394            synchronizer.sync_folders(
395                source_folder=source,
396                dest_folder=destination,
397                now_millis=int(round(time.time() * 1000)),
398                reporter=reporter,
399                encryption_settings_provider=self._b2_encryption(),
400            )

405    def download(self, key: PATH_LIKE) -> Path:
406        source = self.get_b2_sync_url(key)
407        destination = self.local_path / key

409        return destination

411    def upload(self, key: PATH_LIKE) -> str:
412        local_path = self / key
413        if not local_path.is_dir():
414            raise NotADirectoryError(f"Can only upload directories. Got: {key}")
415        if not local_path.exists():
416            raise FileNotFoundError(f"Directory does not exists. Got: {key}")
417        source = str(local_path)
418        assert local_path
419        destination = self.get_b2_sync_url(key)
420        self.sync(source, destination)
421        return destination
